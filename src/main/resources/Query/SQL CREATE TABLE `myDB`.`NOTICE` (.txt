  `NOTICE_YN` VARCHAR(1) NULL COMMENT '공지글 여부',

  `USER_ID` VARCHAR(8) NULL COMMENT '작성자',
  `READ_CNT` INT NULL COMMENT '조회수',
  `REG_ID` VARCHAR(8) NULL COMMENT '최초 등록자아이디',
  `REG_DT` DATETIME NULL DEFAULT SYSDATE() COMMENT '최초 등록시간',
  `CHG_ID` VARCHAR(8) NULL COMMENT '마지막 수정자 아이디',
  `CHG_DT` DATETIME NULL DEFAULT SYSDATE() COMMENT '마지막 수정시간',
  `USER_NAME` VARCHAR(20) NULL COMMENT '닉네임',
  PRIMARY KEY(NOTICE_SEQ)
) ENGINE=InnoDB
